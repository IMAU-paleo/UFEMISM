# Compiler names:
  F90 = mpifort

# Setting the path for the netcdf libraries:  
  NCLIB     = $(shell nf-config --flibs)
  NCINC     = $(shell nf-config --fflags)
  PETSC_DIR = ${HOME}/local_projects/petsc
  PETSC_ARCH= arch-linux-c-debug

# Compiler flags:
  COMPILER_FLAGS_BASIC    = -O3 -J${MOD_PATH} -Wall -ffree-line-length-none -cpp \
                            -Werror=implicit-interface -fimplicit-none -g -march=native
  COMPILER_FLAGS_CHECK    = -fcheck=all -fbacktrace -Og

  F90FLAGS                = ${COMPILER_FLAGS_BASIC}\
                            ${NCINC} -I${PETSC_DIR}/include \
                            -I${PETSC_DIR}/${PETSC_ARCH}/include

# Linking flags of libaries:
  LDFLAGS                 = -L/usr/local/lib ${NCLIB} \
                            -Wl,-rpath=${PETSC_DIR}/${PETSC_ARCH}/lib \
                            ${PETSC_DIR}/${PETSC_ARCH}/lib/libpetsc.so \
                            -lmkl_rt -g
